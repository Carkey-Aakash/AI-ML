def percentile(data, p):
    
    # lets sort data
    sorted_data = sorted(data)

    # now finding index position 0-based index
    n = len(sorted_data)
    pos= (p/100) * (n-1)

    # if pos is decimal we have 2 index lower and upper
    lower= int(pos)
    upper = lower+ 1 

    # Exception case
    if upper >= n:
        return sorted_data[lower]

    # formula to calculate percentile value = lower + fractional *(upper - lower)
    fraction = pos - lower
    return sorted_data[lower] + fraction * (sorted_data[upper] - sorted_data[lower])

def mid_range(data):
    if not data:
        return None 
    #  Let's find minimum and maximum values
    minimum = min(data)
    maximum = max(data)
    
    # Compute mid-range
    mid = (minimum + maximum) / 2
    return mid

def  mean_absolute_deviation(data):
    # if no data then avoids error
    if not data:
        return None
    # finding mean
    mean = sum(data) / len(data)

    # diving in parts to find MAD
    numerator= 0 
    for x in data:
        diff= abs(x-mean)
        numerator+= diff
    return  numerator / len (data)

def percentile(data, p):
    
   # Computing percentile 
    data = sorted(data)
    n = len(data)
    pos = (p / 100) * (n - 1)
    lower = int(pos)
    upper = lower + 1
    if upper >= n:
        return data[lower]
    fraction = pos - lower
    return data[lower] + fraction * (data[upper] - data[lower])

def iqr(data):
    # Compute Interquartile Range i.e using 75-25 percentile
    return percentile(data, 75) - percentile(data, 25)

def standard_deviation(data):
    n = len(data)
    mean = sum(data) / n
    variance = sum((x - mean) ** 2 for x in data) / n
    return variance ** 0.5

def compare_dispersion(data1, data2):
    # Compute SD and IQR for both datasets
    sd1 = standard_deviation(data1)
    sd2 = standard_deviation(data2)
    iqr1 = iqr(data1)
    iqr2 = iqr(data2)

    print(f"Dataset 1: SD={sd1:.2f}, IQR={iqr1:.2f}")  # floating point number and f-string
    print(f"Dataset 2: SD={sd2:.2f}, IQR={iqr2:.2f}")

    # Compare SD
    if sd1 > sd2:
        print("Based on SD: Dataset 1 is more dispersed")
    elif sd2 > sd1:
        print("Based on SD: Dataset 2 is more dispersed")
    else:
        print("Based on SD: Both datasets are equally dispersed")

    # Compare IQR
    if iqr1 > iqr2:
        print("Based on IQR: Dataset 1 is more dispersed")
    elif iqr2 > iqr1:
        print("Based on IQR: Dataset 2 is more dispersed")
    else:
        print("Based on IQR: Both datasets are equally dispersed")

Write a function rolling_std(data, window). Explanation: For a given window size (like 3), compute standard deviation for each rolling segment. Example: data = [10, 12, 14, 20, 22] rolling window = 3 compute std for [10,12,14], [12,14,20], [14,20,22].

def rolling_std(data, window):
    n = len(data)
    if window > n:
        return []  # window too big

    result = []
    for i in range(n - window + 1):  # ensures the window does not go past the end of the list
        segment = data[i: i+window]
        mean = sum(segment) / window
        variance = sum((x - mean) ** 2 for x in segment) / window
        sd = variance ** 0.5
        result.append(sd)
    return result

def detect_high_variability_days(data, threshold):
    high_days = set()  # store high variability days without duplicates
    window = 5

    for i in range(len(data) - window + 1):
        segment = data[i:i+window]
        mean = sum(segment) / window
        sd = (sum((x - mean) ** 2 for x in segment) / window) ** 0.5  # standard deviation
        if sd > threshold:
            high_days.update(range(i, i+window))  # add all days in this window

    return sorted(high_days)

import random
import matplotlib.pyplot as plt

# Step 1: Define rolling_std function
def rolling_std(data, window):
    result = []
    for i in range(len(data) - window + 1):
        segment = data[i:i+window]
        mean = sum(segment)/window
        sd = (sum((x - mean)**2 for x in segment)/window)**0.5
        result.append(sd)
    return result

# Step 2: Create dataset of 40 numbers
